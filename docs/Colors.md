## Colors

### RGBtoHEX
```js
const { RGBtoHEX } = require("sonicware.js");
console.log(RGBtoHEX(255, 255, 255)); // -> #ffffff
```

### HEXtoRGB
```js
const { HEXtoRGB } = require("sonicware.js");
console.log(HEXtoRGB("#FFFFFF")); // -> { r: 255, g: 255, b: 255 }
```
